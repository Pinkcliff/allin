# 核心模块详解

## 模块概述

核心模块包含数据采集、数据存储和数据同步三大功能，是整个系统的数据基础。

## 1. 数据采集模块

### 传感器配置

系统模拟以下传感器:

| 传感器类型 | 数量 | 范围 | 单位 |
|-----------|------|------|------|
| 风扇PWM | 1600 | 0-1000 | - |
| 温度传感器 | 100 | -20~80 | ℃ |
| 风速传感器 | 100 | 0~30 | m/s |
| 温湿度传感器 | 4 | 温度:-20~80, 湿度:0~100 | ℃, % |
| 大气压力传感器 | 1 | 温度:-20~80, 压力:0~100 | ℃, KPa |

采样率: 10次/秒

### 数据生成器

```python
class SensorDataGenerator:
    """传感器数据生成器类"""

    def __init__(self):
        self.cfg = config.SENSOR_CONFIG

    def generate_fans(self) -> list:
        """生成1600个风扇PWM值 (0-1000) - 完全随机"""
        return [random.randint(0, 1000)
                for _ in range(self.cfg['fans']['count'])]

    def generate_temp_sensors(self) -> list:
        """生成100个温度传感器值 (-20~80℃)"""
        return [round(random.uniform(-20, 80), 1)
                for _ in range(self.cfg['temp_sensors']['count'])]

    def generate_wind_speed_sensors(self) -> list:
        """生成100个风速传感器值 (0~30m/s)"""
        return [round(random.uniform(0, 30), 1)
                for _ in range(self.cfg['wind_speed_sensors']['count'])]

    def generate_temp_humidity_sensors(self) -> list:
        """生成4个温湿度传感器值"""
        result = []
        for _ in range(self.cfg['temp_humidity_sensors']['count']):
            result.append({
                'temperature': round(random.uniform(-20, 80), 1),
                'humidity': round(random.uniform(0, 100), 1)
            })
        return result

    def generate_pressure_sensor(self) -> dict:
        """生成大气压力传感器值"""
        return {
            'temperature': round(random.uniform(-20, 80), 1),
            'pressure': round(random.uniform(0, 100), 2)
        }

    def generate_sample(self) -> dict:
        """生成一次完整的采样数据"""
        return {
            'timestamp': time.time(),
            'fans': self.generate_fans(),
            'temp_sensors': self.generate_temp_sensors(),
            'wind_speed_sensors': self.generate_wind_speed_sensors(),
            'temp_humidity_sensors': self.generate_temp_humidity_sensors(),
            'pressure_sensor': self.generate_pressure_sensor()
        }
```

### 数据采集流程

```
定时器触发 (10Hz)
    │
    ▼
生成采样数据
    │
    ├─→ 风扇PWM (1600个)
    ├─→ 温度传感器 (100个)
    ├─→ 风速传感器 (100个)
    ├─→ 温湿度传感器 (4个)
    └─→ 大气压力传感器 (1个)

```

## 2. 数据存储模块

### Redis数据库结构

```
数据结构设计:

1. 采集元数据
   Key: collection:meta:{collection_id}
   Type: Hash
   Fields:
     - id: 采集ID
     - name: 采集名称
     - created_at: 创建时间
     - status: 状态 (collecting/completed)

2. 采集列表
   Key: collections:list
   Type: Set
   Members: collection_id

3. 时间戳集合
   Key: collection:{collection_id}:timestamps
   Type: Sorted Set (按时间戳排序)
   Score: timestamp
   Member: timestamp (作为值)

4. 采样数据
   Key: collection:{collection_id}:data:{timestamp}
   Type: Hash
   Fields:
     - fans: JSON数组
     - temp_sensors: JSON数组
     - wind_speed_sensors: JSON数组
     - temp_humidity_sensors: JSON数组
     - pressure_sensor: JSON对象

5. 实时数据
   Key: realtime:data
   Type: String (JSON)
   TTL: 5秒
```

### Redis操作类

```python
class RedisDatabase:
    """Redis数据库操作类"""

    def __init__(self):
        self.redis_client = redis.Redis(**config.REDIS_CONFIG)
        self._test_connection()

    def create_collection(self, name: str) -> str:
        """创建新的数据采集"""
        collection_id = f"collection_{int(time.time() * 1000)}"
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

        # 存储元数据
        meta_key = f"collection:meta:{collection_id}"
        self.redis_client.hset(meta_key, 'id', collection_id)
        self.redis_client.hset(meta_key, 'name', name)
        self.redis_client.hset(meta_key, 'created_at', timestamp)
        self.redis_client.hset(meta_key, 'status', 'collecting')

        # 添加到采集列表
        self.redis_client.sadd("collections:list", collection_id)

        return collection_id

    def save_sample_data(self, collection_id: str, data: dict) -> bool:
        """保存单次采样数据"""
        try:
            timestamp = data['timestamp']
            ts_key = f"collection:{collection_id}:timestamps"

            # 添加时间戳到有序集合
            self.redis_client.zadd(ts_key, {str(timestamp): timestamp})

            # 存储数据
            data_key = f"collection:{collection_id}:data:{timestamp}"
            self.redis_client.hset(data_key, 'fans',
                                   json.dumps(data['fans']))
            self.redis_client.hset(data_key, 'temp_sensors',
                                   json.dumps(data['temp_sensors']))
            self.redis_client.hset(data_key, 'wind_speed_sensors',
                                   json.dumps(data['wind_speed_sensors']))
            self.redis_client.hset(data_key, 'temp_humidity_sensors',
                                   json.dumps(data['temp_humidity_sensors']))
            self.redis_client.hset(data_key, 'pressure_sensor',
                                   json.dumps(data['pressure_sensor']))

            return True
        except Exception as e:
            print(f"保存数据失败: {e}")
            return False

    def get_collections_list(self) -> list:
        """获取所有采集列表"""
        collection_ids = self.redis_client.smembers("collections:list")
        collections = []

        for cid in collection_ids:
            meta = self.redis_client.hgetall(f"collection:meta:{cid}")
            if meta:
                count = self.redis_client.zcard(f"collection:{cid}:timestamps")
                meta['sample_count'] = str(count)
                collections.append(meta)

        return sorted(collections, key=lambda x: x['created_at'], reverse=True)

    def get_collection_data(self, collection_id: str,
                           start: int = 0, end: int = -1) -> list:
        """获取指定采集的数据"""
        ts_key = f"collection:{collection_id}:timestamps"

        timestamps = self.redis_client.zrange(ts_key, start, end)

        data_list = []
        for ts in timestamps:
            ts_float = float(ts)
            data_key = f"collection:{collection_id}:data:{ts_float}"
            data = self.redis_client.hgetall(data_key)

            if data:
                data_list.append({
                    'timestamp': ts_float,
                    'fans': json.loads(data['fans']),
                    'temp_sensors': json.loads(data['temp_sensors']),
                    'wind_speed_sensors': json.loads(data['wind_speed_sensors']),
                    'temp_humidity_sensors': json.loads(data['temp_humidity_sensors']),
                    'pressure_sensor': json.loads(data['pressure_sensor'])
                })

        return data_list

    def delete_collection(self, collection_id: str) -> bool:
        """删除指定采集"""
        try:
            ts_key = f"collection:{collection_id}:timestamps"
            timestamps = self.redis_client.zrange(ts_key, 0, -1)

            # 删除所有数据
            for ts in timestamps:
                ts_float = float(ts)
                data_key = f"collection:{collection_id}:data:{ts_float}"
                self.redis_client.delete(data_key)

            # 删除时间戳集合和元数据
            self.redis_client.delete(ts_key)
            self.redis_client.delete(f"collection:meta:{collection_id}")
            self.redis_client.srem("collections:list", collection_id)

            return True
        except Exception as e:
            print(f"删除失败: {e}")
            return False
```

## 3. 数据同步模块

### MongoDB数据结构

```
MongoDB集合结构:

1. collections (采集元数据)
   Document:
     {
       "collection_id": "collection_xxx",
       "name": "采集名称",
       "created_at": "2026-04-10 10:00:00",
       "status": "completed",
       "sample_count": 600,
       "synced_at": "2026-04-10 10:01:30"
     }
   索引: {collection_id: 1} (唯一)

2. samples (采样数据)
   Document:
     {
       "collection_id": "collection_xxx",
       "timestamp": 1682345678.123,
       "datetime": "2026-04-10 10:00:00.123",
       "fans": [100, 200, ...],
       "temp_sensors": [25.5, 26.3, ...],
       "wind_speed_sensors": [10.5, 12.3, ...],
       "temp_humidity_sensors": [...],
       "pressure_sensor": {...}
     }
   索引: {collection_id: 1, timestamp: 1} (唯一)
```

### 数据同步类

```python
class MongoSync:
    """MongoDB数据同步类"""

    def __init__(self, mongo_uri: str, db_name: str):
        self.mongo_uri = mongo_uri
        self.db_name = db_name
        self.client = None
        self.redis_db = RedisDatabase()

    def sync_all_collections(self) -> dict:
        """同步所有采集数据到MongoDB"""
        try:
            from pymongo import MongoClient, UpdateOne

            # 连接MongoDB
            self.client = MongoClient(self.mongo_uri)
            db = self.client[self.db_name]
            collections_col = db["collections"]
            samples_col = db["samples"]

            # 创建索引
            collections_col.create_index("collection_id", unique=True)
            samples_col.create_index([("collection_id", 1),
                                     ("timestamp", 1)], unique=True)

            # 获取所有采集
            all_collections = self.redis_db.get_collections_list()

            result = {
                'collections': 0,
                'samples': 0,
                'errors': []
            }

            # 同步每个采集
            for coll in all_collections:
                collection_id = coll['id']

                # 同步元数据
                meta = self.redis_db.get_collection_meta(collection_id)
                sample_count = self.redis_db.get_sample_count(collection_id)

                if meta:
                    doc = {
                        "collection_id": collection_id,
                        "name": meta.get('name', ''),
                        "created_at": meta.get('created_at', ''),
                        "status": meta.get('status', ''),
                        "sample_count": sample_count,
                        "synced_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                    }
                    collections_col.update_one(
                        {"collection_id": collection_id},
                        {"$set": doc},
                        upsert=True
                    )
                    result['collections'] += 1

                # 同步样本数据 (分批)
                if sample_count > 0:
                    offset = 0
                    batch_size = 500

                    while offset < sample_count:
                        data_list = self.redis_db.get_collection_data(
                            collection_id, offset, offset + batch_size - 1
                        )

                        if not data_list:
                            break

                        batch_operations = []
                        for data in data_list:
                            doc = {
                                "collection_id": collection_id,
                                "timestamp": data['timestamp'],
                                "datetime": datetime.fromtimestamp(
                                    data['timestamp']
                                ).strftime("%Y-%m-%d %H:%M:%S.%f")[:-3],
                                "fans": data['fans'],
                                "temp_sensors": data['temp_sensors'],
                                "wind_speed_sensors": data['wind_speed_sensors'],
                                "temp_humidity_sensors": data['temp_humidity_sensors'],
                                "pressure_sensor": data['pressure_sensor']
                            }
                            batch_operations.append(
                                UpdateOne(
                                    {"collection_id": collection_id,
                                     "timestamp": data['timestamp']},
                                    {"$set": doc},
                                    upsert=True
                                )
                            )

                        if batch_operations:
                            samples_col.bulk_write(batch_operations,
                                                 ordered=False)
                            result['samples'] += len(batch_operations)

                        offset += len(data_list)

            return result

        except Exception as e:
            return {
                'success': False,
                'error': str(e)
            }
        finally:
            if self.client:
                self.client.close()
```

## 使用示例

```python
# 示例1: 数据采集
from 核心模块.数据采集 import SensorDataGenerator
from 核心模块.数据存储 import RedisDatabase

generator = SensorDataGenerator()
db = RedisDatabase()

# 创建采集
collection_id = db.create_collection("测试采集")

# 采集数据
for i in range(100):
    data = generator.generate_sample()
    db.save_sample_data(collection_id, data)
    time.sleep(0.1)  # 10Hz

# 标记完成
db.finish_collection(collection_id)

# 示例2: 数据查询
collections = db.get_collections_list()
for coll in collections:
    print(f"{coll['name']}: {coll['sample_count']} 个样本")

    data_list = db.get_collection_data(coll['id'], 0, 9)  # 前10条
    for data in data_list:
        print(f"  {data['timestamp']}: {len(data['fans'])} 个风扇")

# 示例3: MongoDB同步
from 核心模块.数据同步 import MongoSync

sync = MongoSync("mongodb://localhost:27017/", "sensor_data")
result = sync.sync_all_collections()
print(f"同步完成: {result['collections']} 个采集, "
      f"{result['samples']} 条样本")
```

## 性能优化

1. **批量操作**: MongoDB同步使用批量写入 (500条/批)
2. **索引优化**: 创建唯一索引加速查询
3. **分页查询**: Redis使用Sorted Set分页
4. **连接池**: 复用数据库连接
5. **异步处理**: 采集在独立线程运行
